function titleInfo()
{
  const OriginTitle = document.title;
  document.addEventListener('visibilitychange', function() {
  if (document.hidden) {
      document.title = '哎呀，浏览器崩溃啦!';
      document.getElementById("icon").href="../img/o.ico";
  } else {
        document.title = '又好了嘻嘻：)';
